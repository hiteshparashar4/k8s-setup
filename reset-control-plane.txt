#!/bin/bash
set -e

echo "🚫 Installing cni libraries..."
CNI_VERSION="v1.3.0"
ARCH="amd64"
OS="linux"
CNI_DOWNLOAD_URL="https://github.com/containernetworking/plugins/releases/download/${CNI_VERSION}/cni-plugins-${OS}-${ARCH}-${CNI_VERSION}.tgz"

echo "🚫 Turning off swap..."
sudo swapoff -a
sudo sed -i '/ swap / s/^\(.*\)$/#\1/' /etc/fstab

echo "🔁 Resetting kubeadm on control plane..."
sudo kubeadm reset -f --cri-socket unix:///var/run/containerd/containerd.sock

echo "🧹 Cleaning Kubernetes files and directories on control plane..."
sudo rm -rf \
  /etc/cni/net.d \
  /etc/kubernetes \
  /var/lib/etcd \
  /var/lib/kubelet/* \
  /var/lib/cni/ \
  /run/kubernetes \
  /opt/cni \
  $HOME/.kube

echo "🔄 Restarting containerd service..."
sudo systemctl restart containerd

echo "🔥 Flushing iptables..."
sudo iptables -F
sudo iptables -t nat -F
sudo iptables -t mangle -F
sudo iptables -X

echo "🧰 Loading kernel modules..."
sudo modprobe overlay
sudo modprobe br_netfilter

echo "📌 Persisting module load settings..."
cat <<EOF | sudo tee /etc/modules-load.d/k8s.conf
overlay
br_netfilter
EOF

echo "📶 Applying sysctl params..."
cat <<EOF | sudo tee /etc/sysctl.d/k8s.conf
net.bridge.bridge-nf-call-iptables  = 1
net.bridge.bridge-nf-call-ip6tables = 1
net.ipv4.ip_forward                 = 1
EOF

sudo sysctl --system

echo "✅ Control plane node reset complete. Ready for kubeadm init."
